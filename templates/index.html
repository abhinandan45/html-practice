from sklearn.pipeline import Pipeline
import streamlit as st

# 1. Load trained model (Trained elsewhere)
@st.cache_resource
def load_model():
    return pickle.load(open('spam_model.pkl', 'rb'))

pipeline = load_model()

# 2. Get User Input
user_input = st.text_area("Enter Email Content:")

# 3. Predict & Deploy
if st.button('Classify Email', help='Runs prediction on the deployed model'):
    if user_input:
        # Prediction happens inside the deployed Streamlit app
        prediction = pipeline.predict([user_input])[0]
        result = "SPAM ⚠️" if prediction == 1 else "NOT SPAM ✅"
        st.success(f"Classification Result: {result}", unsafe_allow_html=True)
    else:
        st.error("Please enter text for classification.")
